Program to print unique elements in a sorted array



	#include <QCoreApplication>
	#include <stack>
	#include <iostream>
	using namespace std;
	int main(int argc, char *argv[])
	{
		QCoreApplication a(argc, argv);

		int arr[10] = {1,2,2,3,4,5,5,6,6,7};

		std::stack<int> myStack;

		for(int i=0;i<10;i++) {
			if(myStack.empty()){
				cout<<arr[i]<<" ";
				myStack.push(arr[i]);
			}
			if(arr[i] != myStack.top()) {
				cout<<arr[i]<<" ";
				myStack.push(arr[i]);
			}
		}
		return a.exec();
	}
