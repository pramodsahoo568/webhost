Program to demonstrate Map of Vectors operation



	#include <QCoreApplication>
	#include <map>
	#include <vector>
	#include <iostream>

	using namespace std;

	int main(int argc, char *argv[])
	{
		QCoreApplication a(argc, argv);
		//Display Items in Vector
		std::vector<int> vecItems = {1,2,3,4};
		std::vector<int>::iterator it_vecItem;

		for(it_vecItem = vecItems.begin();it_vecItem!=vecItems.end();it_vecItem++) {
			int item = *it_vecItem;
			cout<<item<<" ";
		}

		// Create a Map of vectors
		std::map<int, std::vector<int>> mapVector;
		std::map<int, std::vector<int>>::iterator it_mapVector;
		mapVector.insert(std::make_pair(1,vecItems));


		//Display Map of vector
		cout<<std::endl<<"Map of vector Items"<<std::endl;
		for(it_mapVector=mapVector.begin();it_mapVector!=mapVector.end();it_mapVector++){
			cout<<(*it_mapVector).first<<std::endl;
			 std::vector<int> vec = (*it_mapVector).second;
			for(auto it = vec.begin();it!=vec.end();it++) {
				int item = *it;
				cout<<item<<" ";
			}
		}
		return a.exec();
	}
