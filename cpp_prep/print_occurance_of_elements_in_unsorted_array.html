Program to print occurance of elements in unsorted array



	#include <QCoreApplication>
	#include <iostream>
	#include <map>
	#include <map>


	using namespace std;
	int main(int argc, char *argv[])
	{
		QCoreApplication a(argc, argv);

		std::vector<int> numbers = {1,4,3,1,5,2,4};

		std::map<int,int> unique_map;
		std::map<int,int>::iterator map_iterator;

		std::vector<int>::iterator it_numbers;
		for(it_numbers=numbers.begin();it_numbers!=numbers.end();it_numbers++) {
			int num = *it_numbers;
			map_iterator =  unique_map.find(num);
			if(unique_map.find(num)==unique_map.end()) {
			   unique_map.insert(std::make_pair(num,1));

			}
			else {
				map_iterator->second = map_iterator->second+1;
			}
		 }
		 cout<<"number"<<" : "<<"count"<<std::endl;
		for(map_iterator=unique_map.begin();map_iterator!=unique_map.end();map_iterator++) {
			int number = map_iterator->first;
			int count = map_iterator->second;
			cout<<number<<"     :     "<<count<<std::endl;
		}



		return a.exec();
	}
